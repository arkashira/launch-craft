**2. Unit tests (RSpec style)**  
```ruby
RSpec.describe DirectorySubmissionService do
  let(:product) { create(:product, name: 'Test', url: 'https://test.com', short_description: 'Demo', category: 'Games') }
  let(:directories) { %w[apple_app_store google_play] }

  before do
    allow(DirectoryEndpoint).to receive(:for).and_return(
      'apple_app_store' => { url: 'https://api.apple.com/submit', method: :post },
      'google_play'     => { url: 'https://api.google.com/submit', method: :post }
    )
  end

  it 'creates a submission record for each directory' do
    service = DirectorySubmissionService.new(product, directories)
    expect { service.call }.to change { Submission.count }.by(2)
  end

  it 'sends correct payload to each directory API' do
    service = DirectorySubmissionService.new(product, directories)
    expect(service).to receive(:post_to_api).with('apple_app_store', hash_including(name: 'Test')).once
    expect(service).to receive(:post_to_api).with('google_play', hash_including(url: 'https://test.com')).once
    service.call
  end

  it 'marks status as pending immediately' do
    service = DirectorySubmissionService.new(product, directories)
    service.call
    expect(Submission.where(directory: 'apple_app_store').first.status).to eq('Pending')
  end

  context 'when API returns error' do
    before { allow(service).to receive(:post_to_api).and_raise(StandardError.new('API error')) }

    it 'sets status to Failed and continues with other directories' do
      service = DirectorySubmissionService.new(product, directories)
      service.call
      expect(Submission.where(directory: 'apple_app_store').first.status).to eq('Failed')
      expect(Submission.where(directory: 'google_play').first.status).to eq('Pending')
    end
  end
end
```
