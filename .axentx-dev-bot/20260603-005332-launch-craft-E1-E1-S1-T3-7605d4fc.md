```ruby
# frozen_string_literal: true
# ----------------------------------------------------------------------
# DirectorySubmissionService
#
# * Loads and memoises the YAML configuration.
# * Validates that the requested directory exists.
# * Checks that the caller supplied every field listed in `required_fields`.
# * Builds a JSON payload that contains **exactly** those fields.
# * Logs the attempt (timestamp, directory, payload) via Rails.logger.
# * Returns a lightweight OpenStruct that mimics a real response:
#   status – :pending (real integration will replace this)
#   directory – human readable name
#   payload – JSON string that would be POSTed
#   endpoint – stub URL from the YAML
# ----------------------------------------------------------------------
require 'yaml'
require 'json'
require 'ostruct'

class DirectorySubmissionService
  CONFIG_PATH = Rails.root.join('config', 'directory_endpoints.yml')

  # Load the YAML once and keep it in memory.
  def self.directories
    @directories ||= YAML.safe_load(File.read(CONFIG_PATH), symbolize_names: true)
  end

  # Public API
  #
  # @param directory_key [String, Symbol] e.g. :apple_app_store
  # @param submission_data [Hash] keys may be strings or symbols
  # @return [OpenStruct] with status, directory, payload, endpoint
  # @raise [ArgumentError] if directory not found or required fields missing
  def self.submit(directory_key, submission_data)
    key = directory_key.to_sym
    config = directories[key]
    raise ArgumentError, "Unknown directory: #{directory_key}" unless config

    required_fields = config[:required_fields]
    missing = required_fields.select do |field|
      !submission_data.key?(field) && !submission_data.key?(field.to_s)
    end

    raise ArgumentError, "Missing required fields: #{missing.join(', ')}" unless missing.empty?

    payload_data = {}
    required_fields.each do |field|
      payload_data[field] = submission_data.key?(field) ? submission_data[field] : submission_data[field.to_s]
    end

    payload = JSON.dump(payload_data)
    endpoint = config[:api_endpoint]

    Rails.logger.info({
      event: 'directory_submission_attempt',
      timestamp: Time.current.iso8601,
      directory: config[:name],
      payload_keys: payload_data.keys,
      endpoint: endpoint
    })

    OpenStruct.new(
      status: :pending,
      directory: config[:name],
      payload: payload,
      endpoint: endpoint
    )
  end
end
```
