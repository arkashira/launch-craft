import sqlite3
from contextlib import contextmanager
from pathlib import Path

DB_PATH = Path(__file__).parent / "database.db"

class Database:
    def __init__(self):
        self.conn = sqlite3.connect(DB_PATH, check_same_thread=False)
        self.conn.execute("PRAGMA foreign_keys = ON;")
        self._ensure_schema()

    def _ensure_schema(self):
        with self.conn:
            self.conn.executescript(Path("schema.sql").read_text())

    @contextmanager
    def cursor(self):
        cur = self.conn.cursor()
        try:
            yield cur
            self.conn.commit()
        finally:
            cur.close()

    def submit_product(self, product_name: str, product_description: str, directory_id: int):
        with self.cursor() as cur:
            cur.execute(
                """
                INSERT INTO products (product_name, product_description, directory_id)
                VALUES (?, ?, ?)
                """,
                (product_name, product_description, directory_id),
            )
            return cur.lastrowid

    def get_product(self, product_id: int):
        with self.cursor() as cur:
            cur.execute(
                """
                SELECT id, product_name, product_description, directory_id, review_status, created_at
                FROM products
                WHERE id = ?
                """,
                (product_id,),
            )
            return cur.fetchone()

    def update_review_status(self, product_id: int, status: str):
        if status not in {"pending", "approved", "rejected"}:
            raise ValueError("Invalid review status")
        with self.cursor() as cur:
            cur.execute(
                """
                UPDATE products
                SET review_status = ?
                WHERE id = ?
                """,
                (status, product_id),
            )
