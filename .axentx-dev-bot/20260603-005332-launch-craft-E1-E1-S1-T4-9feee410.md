# app/models/submission.rb
class Submission < ApplicationRecord
  belongs_to :product
  belongs_to :directory

  validates :status, presence: true

  enum status: {
    pending: 0,
    submitted: 1,
    approved: 2,
    rejected: 3,
    failed: 4
  }

  after_create :start_submission_process

  private

  def start_submission_process
    # This will trigger the submission workflow
    process_submission
  end

  def process_submission
    if directory.api?
      submit_via_api
    else
      submit_via_email
    end
  rescue StandardError => e
    update(status: :failed, error: e.message)
  end

  def submit_via_api
    begin
      directory.api.submit(self)
      update(status: :submitted)
    rescue => e
      update(status: :failed, error: e.message)
    end
  end

  def submit_via_email
    begin
      directory.email_template.send(self)
      update(status: :submitted)
    rescue => e
      update(status: :failed, error: e.message)
    end
  end
end
```

**Key Improvements**:
- **Status Enum Expansion**: Added `failed` state for error handling.
- **Error Tracking**: Logs errors and updates status automatically.
- **Asynchronous Processing**: Uses `after_create` to trigger processing.
