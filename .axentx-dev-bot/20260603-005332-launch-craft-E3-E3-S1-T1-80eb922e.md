# spec/controllers/public/products_controller_spec.rb
require 'rails_helper'

RSpec.describe Public::ProductsController, type: :controller do
  let(:product) { create(:product, slug: 'awesome-gadget', name: 'Awesome Gadget', description: 'Cool stuff', screenshots: ['s1.png', 's2.png']) }
  let(:review1) { create(:review, product: product, source: 'amazon', rating: 5, content: 'Great!') }
  let(:review2) { create(:review, product: product, source: 'ebay', rating: 4, content: 'Good value') }

  before { [review1, review2] }

  describe 'GET #show' do
    context 'when product exists' do
      it 'returns http success' do
        get :show, params: { slug: product.slug }
        expect(response).to have_http_status(:ok)
      end

      it 'renders product name and description' do
        get :show, params: { slug: product.slug }
        expect(response.body).to include(product.name)
        expect(response.body).to include(product.description)
      end

      it 'includes JSON‑LD script with correct @type' do
        get :show, params: { slug: product.slug }
        json_ld = Nokogiri::HTML(response.body).at_css('script[type="application/ld+json"]').text
        data = JSON.parse(json_ld)
        expect(data['@type']).to eq('Product')
        expect(data['name']).to eq(product.name)
        expect(data['description']).to eq(product.description)
        expect(data['image']).to match_array(product.screenshots)
      end

      it 'aggregates reviews from all sources' do
        get :show, params: { slug: product.slug }
        expect(response.body).to include(review1.content)
        expect(response.body).to include(review2.content)
      end
    end

    context 'when product does not exist' do
      it 'returns 404' do
        get :show, params: { slug: 'non-existent' }
        expect(response).to have_http_status(:not_found)
      end
    end

    context 'when product has no reviews' do
      it 'shows empty review section gracefully' do
        product.reviews.destroy_all
        get :show, params: { slug: product.slug }
        expect(response.body).to include('No reviews yet')
      end
    end
  end
end
