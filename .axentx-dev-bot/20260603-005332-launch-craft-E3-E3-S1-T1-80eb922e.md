"""Utility helpers for launch_craft."""
import json
from pathlib import Path

# Resolve the data directory relative to the repository root
DATA_DIR = Path(__file__).resolve().parents[1] / "data"
DATA_FILE = DATA_DIR / "submissions.json"


def ensure_data_file() -> None:
    """Create the data directory and an empty JSON file if they do not exist."""
    DATA_DIR.mkdir(parents=True, exist_ok=True)
    if not DATA_FILE.exists():
        DATA_FILE.write_text("[]", encoding="utf-8")


def load_submissions() -> list[dict]:
    """Read the JSON file and return the list of stored submissions."""
    ensure_data_file()
    with DATA_FILE.open("r", encoding="utf-8") as f:
        return json.load(f)


def save_submission(name: str, description: str) -> None:
    """Append a new submission to the JSON store."""
    submissions = load_submissions()
    submissions.append({"name": name, "description": description})
    with DATA_FILE.open("w", encoding="utf-8") as f:
        json.dump(submissions, f, indent=2, ensure_ascii=False)
```
