RSpec.describe PaymentsController, type: :controller do
  let(:user) { create(:user) }
  let(:submission) { create(:submission, user: user, status: 'free') }

  describe 'POST #create' do
    before { sign_in user }

    context 'when submission is free' do
      it 'creates a Stripe Checkout session with correct amount' do
        expect(Stripe::Checkout::Session).to receive(:create)
          .with(hash_including(
            line_items: [{ price_data: { unit_amount: 2900, currency: 'usd' } }]
          )).and_return(double(id: 'cs_test_123', url: 'https://checkout.stripe.com/pay/cs_test_123'))

        post :create, params: { submission_id: submission.id }

        expect(response).to redirect_to('https://checkout.stripe.com/pay/cs_test_123')
      end
    end

    context 'when submission is already priority' do
      before { submission.update(status: 'priority') }

      it 'does not create a new session' do
        expect(Stripe::Checkout::Session).not_to receive(:create)
        post :create, params: { submission_id: submission.id }
        expect(response).to redirect_to(submission_path(submission))
        expect(flash[:alert]).to eq('Submission already priority.')
      end
    end
  end

  describe 'POST #webhook' do
    let(:session_id) { 'cs_test_123' }
    let(:payload) do
      {
        type: 'checkout.session.completed',
        data: {
          object: {
            id: session_id,
            metadata: { submission_id: submission.id.to_s }
          }
        }
      }.to_json
    end
    let(:sig_header) { 't=12345,v1=abc' }

    before do
      allow(Stripe::Webhook).to receive(:construct_event)
        .and_return(JSON.parse(payload))
      allow(Stripe::Checkout::Session).to receive(:retrieve)
        .with(session_id).and_return(double(id: session_id))
    end

    it 'updates submission status to priority' do
      post :webhook, params: payload, headers: { 'Stripe-Signature' => sig_header }
      submission.reload
      expect(submission.status).to eq('priority')
    end

    it 'sends confirmation email' do

      mail = ActionMailer::Base.deliveries.last
      expect(mail.subject).to eq('Priority Submission Confirmed')
      expect(mail.to).to include(user.email)
    end
  end
