# app/controllers/payments_controller.rb
class PaymentsController < ApplicationController
  before_action :authenticate_user!
  before_action :authorize_submission

  def upgrade
    @submission = current_user.submissions.find(params[:submission_id])
    if @submission.status == 'pending'
      session[:stripe_session_id] = create_checkout_session
      redirect_to checkout_path(session[:stripe_session_id])
    else
      redirect_to submission_path(@submission), alert: 'Already a priority submission'
    end
  end

  private

  def create_checkout_session
    price_id = ENV['STRIPE_PRICE_ID_PRIORITY']
    session[:stripe_session_id] = Stripe::Checkout::Session.create(
      payment_method_types: ['card'],
      line_items: [{ price: price_id, quantity: 1 }],
      mode: 'payment',
      success_url: Rails.application.routes.url_helpers.success_url(request.host, params[:submission_id]),
      cancel_url: Rails.application.routes.url_helpers.cancel_url(request.host, params[:submission_id]),
      metadata: {
        submission_id: @submission.id.to_s,
        user_id: current_user.id.to_s
      }
    )
  end

  def authorize_submission
    redirect_to root_path, alert: 'Not authorized' unless current_user.submissions.exists?(id: params[:submission_id])
  end
end
```
