def test_render_json_ld():
    product = ProductFactory(slug="cool-gadget")
    rendered = render_template("public/products/show.html.erb", product=product)
    json_ld = extract_script_tag(rendered, "application/ld+json")
    data = json.loads(json_ld)
    assert data["@type"] == "Product"
    assert data["name"] == product.name
    assert data["description"] == product.description
    assert data["image"] == product.screenshot_urls[0]
    assert "offers" in data

def test_review_aggregation():
    product = ProductFactory(slug="cool-gadget")
    ReviewFactory(product=product, source="Amazon", rating=5, comment="Great!")
    ReviewFactory(product=product, source="Etsy", rating=4, comment="Good.")
    rendered = render_template("public/products/show.html.erb", product=product)
    reviews = extract_reviews(rendered)
    assert len(reviews) == 2
    assert reviews[0]["source"] == "Amazon"  # newest first

def test_no_reviews():
