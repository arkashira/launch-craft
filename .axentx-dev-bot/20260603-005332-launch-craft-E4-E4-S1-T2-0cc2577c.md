    @directory_traffics = DirectoryTraffic.includes(:directory).all
    respond_to do |format|
      format.csv { send_data generate_csv(@directory_traffics), filename: "directory_traffics-#{Date.today}.csv" }

    CSV.generate(headers: true) do |csv|

      directory_traffics.each do |dt|
        csv << [
          dt.directory.name,
          dt.traffic_count,
          dt.created_at,
          dt.updated_at
        ]
      end
    end
  end
end
