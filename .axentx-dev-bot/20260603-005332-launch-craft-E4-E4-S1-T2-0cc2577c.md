class DirectoryTrafficsController < ApplicationController
  def index
    @directory_traffics = DirectoryTraffic.includes(:directory).all
  end

  def export_csv
    @directory_traffics = DirectoryTraffic.includes(:directory).all
    respond_to do |format|
      format.csv { send_data generate_csv(@directory_traffics), filename: "directory_traffics-#{Date.today}.csv" }
    end
  end

  private

  def generate_csv(directory_traffics)
    CSV.generate(headers: true) do |csv|
      csv << ["Directory Name", "Traffic Count", "Created At", "Updated At"]
      directory_traffics.each do |dt|
        csv << [
          dt.directory.name,
          dt.traffic_count,
          dt.created_at,
          dt.updated_at
        ]
      end
    end
  end
end
```
